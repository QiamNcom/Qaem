function goStep2(){

  const vals=[];

  for(let i=0;i<5;i++){

    if(v) vals.push(v);

  }

  if(vals.length<3){

    alert(lang==='ar'?'رجاءً اذكر/ي ٣ أمثلة على الأقل':'Please list at least 3 examples');

  }

  const tbl=document.getElementById('feelTable');

  tbl.innerHTML='<tr><th id="colA">'+T[lang].colA+'</th><th id="colB">'+T[lang].colB+'</th></tr>';

    const row=tbl.insertRow(-1);

    inp.type='text'; inp.style.width='100%';

    row.insertCell(1).appendChild(inp);

  });

  document.getElementById('step2').classList.remove('hidden');

}
